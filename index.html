# AI Automation Pipeline
from openai import OpenAI
import n8n_client

client = OpenAI()

def automate_workflow(task):
    response = client.chat.completions.create(
        model="gpt-4o",
        messages=[{"role": "user",
                   "content": task}]
    )
    return response.choices[0].message.content

# ✅ Automatización activa — ahorrando tiempo
